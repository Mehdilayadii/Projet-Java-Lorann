17  
18  public class EventsManager {
19  
20                      /**** ATTRIBUTE ****/
21  	/** the model*/
22      private IModel model;
23      
24      /** the view*/
25      private IView view;
26  
27      /** the facing look */
28      private Point player_facing_during_casting;
29      
30      /** the spell's move*/
31      private Point spell_move;
32      
33      /** the player's move*/
34      private Point player_move;
35      
36      /** the check if player is casting spell */
37      private boolean player_casting_spell;
38      
39      /** the check if spell is alive*/
40      private boolean spell_is_alive;
41  
42      public Events event;
43  
44      				/**** CONSTRUCTOR ****/
45      
46      /**
47       * Initiate the Events Manager
48       * @param model the model
49       * @param view the view
50       */
51      public EventsManager(IModel model, IView view) {
52          this.model = model;
53          this.view = view;
54  
55          this.player_facing_during_casting = new Point(0,0);
56          this.spell_move = new Point(0,0);
57          this.player_move = new Point(0,0);
58          this.player_casting_spell = false;
59          this.spell_is_alive = false;
60  
61          this.event = new Events( model, view, player_move.x, player_move.y);
62      }
63  
64      				/**** GETTERS and SETTERS ****/
65      
66      /**
67       * Set the spell state
68       * Player state is a boolean :
69       * @see model.IModel#spellAlive()
70       * - Get the current state
71       */
72      public void setSpell_is_alive(){
73          spell_is_alive = model.spellAlive();
74      }
75  
76      /**
77       * Set the player movement from user inputs, it's also the player look direction.
78       * @see view.IView#return_deplacement_player()
79       * - Get the current deplacement
80       * */
81      public void setPlayer_move() {
82          player_move = view.return_deplacement_player(); //
83      }
84  
85      /**
86       * Get the player movement.
87       * @return Vector of the player movement
88       */
89      public Point getPlayer_move() {
90          return player_move;
91      }
92  
93      				/**** METHODS ****/
94  
95      /**
96       * Set the direction of the player. Same as setPlayerMove() but is save if player stop move. Make able the player to shoot then he don't move.
97       */
98      public void setPlayer_facing_during_casting() {
99          if ((player_move.x != 0) || (player_move.y != 0)) { // player is moving
100             if (spell_is_alive == false) { // spell not exist
101                 player_facing_during_casting.x = player_move.x; // set new facing (x)
102                 player_facing_during_casting.y = player_move.y; // set new facing (y)
103             }
104         }
105     }
106 
107     /**
108      * Change the player location from the user inputs.
109      * Check if player finish (Die or Win) or if he can't reach the point.
110      *
111      * @return False if the player finish.
112      *
113      * @see Events#setFuture_player(int, int)
114      * @see Events#playerCanReach()
115      * @see Events#isGameEnd()
116      */
117     public boolean movePlayer() {
118         event.setFuture_player(player_move.x, player_move.y);
119         if (event.isGameEnd() == true) {
120             return false;
121         }
122         else if (event.playerCanReach()) {
123             model.movePlayer(player_move.x,player_move.y);
124         }
125         return true;
126     }
127 
128     /**
129      * Check if the player pick up the spell. If so make sure that the player haven't been erase by the spell.
130      * @see IModel#deleteSpell()
131      * @see IModel#createElement(int, int, Types)
132      */
133     public void checkPlayerGetSpell() {
134         setSpell_is_alive();
135         if ((spell_is_alive == true) && (model.getSpellLocation().x == model.getPlayerLocation().x) && (model.getSpellLocation().y == model.getPlayerLocation().y)) {
136             model.deleteSpell();
137             model.createElement(model.getPlayerLocation().x,model.getPlayerLocation().y,Types.PLAYER);
138         }
139     }
140 
141     /**
142      * Check if a enemy pick up the spell. If so make kill the good one and delete the spell.
143      * @see IModel#deleteSpell()
144      * @see IModel#killEnemy(int, int)
145      */
146     public void checkMobGetSpell() {
147         setSpell_is_alive();
148         if ((spell_is_alive == true) && model.isThereEnemy(model.getSpellLocation().x,model.getSpellLocation().y)) {
149             model.killEnemy(model.getSpellLocation().x,model.getSpellLocation().y);
150             model.deleteSpell();
151         }
152     }
153 
154     /**
155      * Check if a enemy eat the player. If so make kill the good one and delete the spell.
156      * @see IModel#deleteSpell()
157      * @see IModel#killEnemy(int, int)
158      * @return true if mob get a player
159      */
160     public boolean checkMobGetPlayer() {
161         if (model.isThereEnemy(model.getPlayerLocation().x,model.getPlayerLocation().y)) {
162             view.displayMessage("Game over !");
163             return false;
164         }
165         return true;
166     }
167 
168     /**
169      * Create a spell in function of the player position and his facing.
170      * @see IView#return_casting_player()
171      * @see Events#canCreateSpell(int, int)
172      * @see IModel#createElement(int, int, Types)
173      */
174     public void createSpell() {
175         spell_is_alive = model.spellAlive();
176         player_casting_spell = view.return_casting_player(); // is player casting ?
177         if ((player_casting_spell == true) && (spell_is_alive == false)) { // Player is able to cast spell
178             spell_move.x = player_facing_during_casting.x;
179             spell_move.y = player_facing_during_casting.y;
180             if ((event.canCreateSpell(spell_move.x, spell_move.y)) && ((spell_move.x != 0) || (spell_move.y != 0))) { // player have facing
181                 model.createElement(spell_move.x, spell_move.y,Types.SPELL); // Create spell
182             }
183         }
184     }
185 
186     /**
187      * Move the spell (if exist)
188      * @see Events#setFuture_spell(int, int)
189      * @see IModel#moveSpell(int, int)
190      */
191     public void moveSpell() {
192         if (spell_is_alive == true) { // spell exist
193             event.setFuture_spell(spell_move.x, spell_move.y); // update spell location
194             if (event.spellCanReach() == true) { // there is void (can move)
195                 model.moveSpell(spell_move.x, spell_move.y); // moving spell
196             }
197             else if (event.spellCanReach() == false) { // there is an obstacle
198                 spell_move.x = - spell_move.x; // spell rebound (x)
199                 spell_move.y = - spell_move.y; // spell rebound (y)
200             }
201         }
202     }
203 }
