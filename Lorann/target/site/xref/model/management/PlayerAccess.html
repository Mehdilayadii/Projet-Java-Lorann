4   import model.elements.Mobile.Player;

6   public class PlayerAccess {
7   
8       private MapModel map;
9   
10      //CONSTRUCTOR//
11      public PlayerAccess(MapModel map) {
12          this.map = map;
13      }
14  
15      //METHODS//
16      /**
17       * Move player
18       * @see model.MapModel#getPlayer()
19       * - Get infos about player
20       * @see model.elements.Mobile.Mobile#getLocation()
21       * - Get location of an element
22       * @see model.elements.Mobile.Mobile#setLocation(int ,int)
23       * - Set location of an element
24       * @see model.MapModel#moveElement(int oldX, int oldY, int newX, int newY)
25       * - Move element from current coordinate to new ones
26       * @param moveX Relative direction in X coordinate
27       * @param moveY Relative direction in Y coordinate
28       *
29       */
30      public void movePlayer(int moveX, int moveY) {
31          int newX = map.getPlayer().getLocation().x + moveX;
32          int newY = map.getPlayer().getLocation().y - moveY;
33  
34          map.moveElement(map.getPlayer().getLocation().x,map.getPlayer().getLocation().y,newX,newY);
35          map.getPlayer().setLocation(newX,newY);
36      }
37  
38      /**
39       * Create the player at a special location.
40       * @param x spawn location
41       * @param y spawn location
42       */
43      public void createPlayer(int x, int y) {
44          Player player = new Player("L1",x,y);
45          map.setPlayer(player);
46          map.addElement(player,x,y);
47      }
48  }
